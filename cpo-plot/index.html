Plotting

The orientation distribution function (ODF; normalized expansion series) can be plotted as follows:

import numpy as np
import matplotlib.pyplot as plt
from specfabpy import specfab as sf
from specfabpy import plotting as sfplt

lm, nlm_len = sf.init(6) # initialize specfab with L=6

### Synthetic CPO to be plotted
a2 = np.diag([0,0,1]) # CPO characterized by a^(2)
nlm = sf.a2_to_nlm(a2) # vector of expansion coefficients

### Setup axes + projection
geo, prj = sfplt.getprojection(rotation=45, inclination=45) # load projection
ax = plt.subplot(111, projection=prj)
ax.set_global() # ensure entire S^2 is shown

### Plot
sfplt.plotODF(nlm, lm, ax, lvlset='iso-up') # plot distribution (see src/specfabpy/plotting.py for API)
sfplt.plotcoordaxes(ax, geo) # plot coordinate axes (see src/specfabpy/plotting.py for API)
plt.savefig('ODF-plot.png', dpi=175)
